Para todas as questões abaixo, utilize o modo de comparação do Timer A.

1.Para os itens abaixo, confira a diferença no brilho do LED.
(a) Pisque o LED no pino P1.6 numa frequência de 100 Hz e ciclo de trabalho de 25%. 

#include<msp430g2553.h>
#define LED BIT6

int main(void)
{
	WDTCTL  = WDTPW + WDTHOLD;
	BCSCTL1 = CALBC1_1MHZ;
	DCOCTL  = CALDCO_1MHZ;
	P1DIR  |= LED;
	P1SEL  |= LED;
	P1SEL2 &= ~LED;
	TACCR0  = 10000-1;                //como frequencia é 1Mhz , para 100hz, temos 10^6/100=x , logo x=10^4
	TACCR1  = 25000-1;
	TACCTL1 =OUTMOD_7;              //o ciclo de trabalho é calculado fazendo TACCR1/TACCR0=cicotrabalho/100
	TACTL = TASSEL_2 + ID_0 + MC_1;
	_BIS_SR(LPM0_bits);
	return 0;	
}
